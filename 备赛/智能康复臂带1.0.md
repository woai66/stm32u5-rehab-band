### 3. 姿态融合算法（AHRS）

使用Madgwick算法融合加速度+陀螺仪数据，获取四元数姿态：

```c
// 四元数姿态
float q[4] = {1, 0, 0, 0};  // 初始姿态

// Madgwick姿态更新
void updateAttitude(float ax, float ay, float az,
                    float gx, float gy, float gz) {
    float dt = 0.01f;  // 10ms

    // 归一化加速度
    float norm = sqrtf(ax*ax + ay*ay + az*az);
    ax /= norm; ay /= norm; az /= norm;

    // 四元数更新
    float q0 = q[0], q1 = q[1], q2 = q[2], q3 = q[3];

    float f1 = 2*(q1*q3 - q0*q2) - ax;
    float f2 = 2*(q0*q1 + q2*q3) - ay;
    float f3 = 2*(0.5 - q1*q1 - q2*q2) - az;

    // 梯度下降优化（简化版）
    float Jt[3][4];
    Jt[0][0] = -2*q2; Jt[0][1] = 2*q3; Jt[0][2] = -2*q0; Jt[0][3] = 2*q1;
    Jt[1][0] = 2*q1;  Jt[1][1] = 2*q0;  Jt[1][2] = 2*q3;  Jt[1][3] = 2*q2;
    Jt[2][0] = 0;     Jt[2][1] = -4*q1; Jt[2][2] = -4*q2; Jt[2][3] = 0;

    // 更新四元数
    q[0] += 0.5 * (-q1*gx - q2*gy - q3*gz) * dt;
    q[1] += 0.5 * ( q0*gx + q2*ay - q3*az) * dt;
    q[2] += 0.5 * (-q0*gy + q1*az + q3*gx) * dt;
    q[3] += 0.5 * ( q0*az - q1*ay + q2*gx) * dt;

    // 归一化
    norm = sqrtf(q[0]*q[0] + q[1]*q[1] + q[2]*q[2] + q[3]*q[3]);
    q[0] /= norm; q[1] /= norm; q[2] /= norm; q[3] /= norm;
}
```

### 4. 相对姿态计算（核心创新）

通过两个传感器的四元数计算肘关节角度：

```c
// 四元数乘法
void quatMultiply(float* a, float* b, float* result) {
    result[0] = a[0]*b[0] - a[1]*b[1] - a[2]*b[2] - a[3]*b[3];
    result[1] = a[0]*b[1] + a[1]*b[0] + a[2]*b[3] - a[3]*b[2];
    result[2] = a[0]*b[2] - a[1]*b[3] + a[2]*b[0] + a[3]*b[1];
    result[3] = a[0]*b[3] + a[1]*b[2] - a[2]*b[1] + a[3]*b[0];
}

// 四元数求共轭
void quatConjugate(float* q, float* result) {
    result[0] = q[0];
    result[1] = -q[1];
    result[2] = -q[2];
    result[3] = -q[3];
}

// 计算肘关节角度
float calculateElbowAngle(float* q_upper, float* q_wrist) {
    // q_relative = q_wrist * conjugate(q_upper)
    float q_conj[4], q_rel[4];
    quatConjugate(q_upper, q_conj);
    quatMultiply(q_wrist, q_conj, q_rel);

    // 四元数转欧拉角（Pitch即为肘关节屈伸角度）
    float pitch = asinf(2*(q_rel[0]*q_rel[2] - q_rel[3]*q_rel[1]));
    float angle = pitch * 180.0f / 3.14159f;

    // 归一化到 0°-180° 范围
    if (angle < 0) angle += 360;
    if (angle > 180) angle = 360 - angle;

    return angle;  // 0°=伸直, 150°=完全弯曲
}
```

### 7. 运动质量评分

```c
typedef struct {
    float completion;   // 完成度 0-100
    float smoothness;   // 平稳度 0-100
    float frequency;    // 频率 0-100
    float total_score;  // 综合评分
} RehabScore_t;

// 评分计算
RehabScore_t calculateScore(int action, float* angles, int action_count) {
    RehabScore_t score;

    // 1. 完成度：实际角度 / 标准角度
    float standard_angle = getStandardAngle(action);
    float actual_angle = getMaxAngle(angles);
    score.completion = min(100, (actual_angle / standard_angle) * 100);

    // 2. 平稳度：基于角速度方差
    float gyro_variance = calculateVariance(gyro_buffer, 100);
    score.smoothness = max(0, (1 - gyro_variance / 50.0f) * 100);

    // 3. 频率：动作次数 / 标准次数
    score.frequency = min(100, (action_count / 20.0f) * 100);

    // 4. 综合评分
    score.total_score = score.completion * 0.6 +
                        score.smoothness * 0.3 +
                        score.frequency * 0.1;

    return score;
}

// 动作标准角度
float getStandardAngle(int action) {
    switch(action) {
        case ACTION_ELBOW_FLEX:    return 120.0f;  // 肘关节弯曲
        case ACTION_ARM_ABDUCTION: return 90.0f;   // 手臂外展
        case ACTION_SHOULDER_LIFT: return 150.0f;  // 肩上举
        case ACTION_WRIST_ROT:     return 90.0f;   // 腕关节旋转
        default:                   return 100.0f;
    }
}
```

### 10. 触摸驱动（GT911电容触摸）

```c
#include "gt911.h"
#include "i2c.h"

// 触摸坐标结构
typedef struct {
    uint16_t x;
    uint16_t y;
    uint8_t  state;  // 0=释放, 1=按下
} TouchPoint_t;

// GT911初始化
int8_t gt911_init(void) {
    uint8_t cfg[184] = {0};
    // 读取配置
    if (GT911_ReadReg(0x8047, cfg, 184) != 0) {
        return -1;  // 初始化失败
    }
    return 0;
}

// 读取触摸点
uint8_t gt911_get_points(TouchPoint_t *points, uint8_t max_points) {
    uint8_t buffer[40] = {0};
    uint8_t num = 0;

    // 读取触摸状态
    GT911_ReadReg(0x814E, buffer, 1);
    num = buffer[0] & 0x0F;

    if (num > 0 && num <= max_points) {
        // 读取触摸点坐标
        GT911_ReadReg(0x8150, buffer, num * 8);

        for (uint8_t i = 0; i < num; i++) {
            points[i].x = (buffer[i*8 + 1] << 8) | buffer[i*8];
            points[i].y = (buffer[i*8 + 3] << 8) | buffer[i*8 + 2];
            points[i].state = 1;
        }

        // 清除触摸标志
        buffer[0] = 0;
        GT911_WriteReg(0x814E, buffer, 1);
    }

    return num;
}

// LVGL触摸输入设备初始化
void lv_port_indev_init(void) {
    static lv_indev_data_t indev_data;

    // 配置触摸输入设备
    indev_data.type = LV_INDEV_TYPE_POINTER;
    indev_data.read_cb = touch_read_cb;

    lv_indev_add(&indev_data);
}

// LVGL触摸读取回调
bool touch_read_cb(lv_indev_data_t *data) {
    static TouchPoint_t last_point = {0, 0, 0};
    TouchPoint_t point;

    uint8_t num = gt911_get_points(&point, 1);

    if (num > 0) {
        data->point.x = point.x;
        data->point.y = point.y;
        data->state = LV_INDEV_STATE_PR;
        last_point = point;
    } else {
        data->point.x = last_point.x;
        data->point.y = last_point.y;
        data->state = LV_INDEV_STATE_REL;
    }

    return false;
}

        {20, 460},       // 左下
        {300, 460}       // 右下
    };

    // 依次提示用户点击4个点
```

```dart
        .addHandler(_router);

    _server = await shelf_io.serve(handler, '0.0.0.0', 8080);
    print('Web服务器: http://192.168.1.X:8080');
  }

  Response _router(Request request) {
    if (request.url.path == '') {
      return Response.ok(_getHTML());
    } else if (request.url.path == 'api/data') {
      return Response.ok(
        jsonEncode({
          'today_score': 85,
          'history': [72, 75, 78, 82, 85],
          'timestamp': DateTime.now().toString()
        }),
        headers: {'Content-Type': 'application/json'},
      );
    }
    return Response.notFound('Not found');
  }

  String _getHTML() {
    return '''
    <!DOCTYPE html>
    <html>
    <head><title>康复数据</title></head>
    <body>
      <h1>AI-RehabBand 康复数据</h1>
      <div id="data"></div>
      <script>
        setInterval(() => {
          fetch('/api/data').then(r => r.json()).then(d => {
            document.getElementById('data').innerText = d.today_score;
          });
        }, 1000);
      </script>
    </body>
    </html>
    ''';
  }
}
```
