 方案A：入门级（推荐） ⭐⭐⭐

  核心思路

  简化方向：保留医疗场景 + 去掉云平台 + 算法降级

  项目名称：智能康复手表 —— 脑卒中患者上肢运动评估

  硬件配置（简化版）

  STM32U575 (NUCLEO开发板即可)
      ↓
  LSM6DSV16X (6轴加速度+陀螺仪) ← 只用1个传感器
      ↓
  NUCLEO自带的屏幕显示实时数据
      ↓
  蓝牙模块连接手机App
      ↓

  不做的事：
  - ❌ 自制PCB（用开发板）

  - ✅ 每日训练进度记录
  - ✅ 工程化（自制PCB or 功能完整的开发板）

  核心创新点（3个）

  创新1：康复动作自动识别
  使用 STM32CubeAI 部署轻量CNN（<50KB）

  输入：加速度序列
  输出：[标准握拳, 前臂转动, 手臂外展, 步行, ...]

  模型训练：
    用开源数据集 + 自己录制的动作视频（20份）
    → 用 Edge Impulse 或 TensorFlow Lite 训练
    → 导出到MCU（完全本地推理）

  创新2：康复进度量化
  每次动作评分：
  - 完成度 (0-100%)：动作幅度是否达到标准
  - 平稳度 (0-100%)：运动轨迹是否平稳
  - 频率 (次/分)：运动速度

  总分 = 完成度*60% + 平稳度*30% + 频率*10%

  每日汇总 → 显示周趋势图

  创新3：智能反馈
  如果动作评分<60% → 振动提醒
  显示提示："前臂外展不足，请继续加大幅度"

  患者通过改进 → 评分逐渐提高 → 游戏化体验

  固件架构（FreeRTOS）

  STM32固件任务划分：
  ┌─ Task1: 传感器采集 (100Hz)
  │  └─ LSM6DSV16X → DMA → 环形缓冲区
  │
  ├─ Task2: AI推理 (10Hz)
  │  └─ 每100ms推理一次 → 识别动作
  │
  ├─ Task3: 显示刷新 (10Hz)
  │  └─ OLED屏幕更新评分
  │
  ├─ Task4: 蓝牙通信 (1Hz)
  │  └─ 发送今日数据到App
  │
  └─ Task5: 功耗管理 (1Hz)
     └─ 动态调整MCU频率 (160MHz→24MHz)

  这样设计的好处：
  ✓ 各任务独立，易调试
  ✓ 优先级明确（传感器 > 推理 > UI）
  ✓ 功耗可控（闲置时降频）

      HAL_Init();
      SystemClock_Config();

      // 创建任务
      xTaskCreate(vSensorTask, "Sensor", 256, NULL, 9, &xSensorTask);
      xTaskCreate(vAITask, "AI", 512, NULL, 8, &xAITask);
      xTaskCreate(vDisplayTask, "Display", 256, NULL, 5, &xDisplayTask);

  // 传感器采集任务（100Hz）
  void vSensorTask(void *pvParameters) {
      TickType_t xLastWakeTime = xTaskGetTickCount();

      while(1) {
          // 从LSM6DSV16X读取加速度数据
          float accel[3];
          LSM6DSV16X_GetAccel(accel);  // {x, y, z}
    
          // 放入环形缓冲区
          pushToBuffer(accel);
    
          // 10ms周期采集
          vTaskDelayUntil(&xLastWakeTime, pdMS_TO_TICKS(10));
      }
  }

  // AI推理任务（10Hz）
  void vAITask(void *pvParameters) {
      TickType_t xLastWakeTime = xTaskGetTickCount();

      while(1) {
          // 从缓冲区提取100个样本（1秒数据）
          float features[100];
          getFromBuffer(features, 100);
    
          // 运行AI模型推理
          float output[5];  // 5个动作的概率
          run_classifier(features, output);
    
          // 找到最高概率的动作
          int best_action = argmax(output, 5);
          float confidence = output[best_action];
    
          if (confidence > 0.7) {
              // 识别成功，计算评分
              calculateScore(best_action);
          }
    
          // 100ms推理一次
          vTaskDelayUntil(&xLastWakeTime, pdMS_TO_TICKS(100));
      }
  }

  // 屏幕刷新任务（10Hz）
  void vDisplayTask(void *pvParameters) {
      while(1) {

  #include "lsm6dsv16x_reg.h"

  typedef struct {
      float accel[3];  // x, y, z
      float gyro[3];   // gx, gy, gz
      uint32_t timestamp;
  } sensor_data_t;

  // 环形缓冲区
  #define BUFFER_SIZE 100
  sensor_data_t sensor_buffer[BUFFER_SIZE];
  uint16_t write_idx = 0;

  // 读取传感器数据
  void readSensor(void) {
      int16_t raw_accel[3], raw_gyro[3];

      // I2C读取原始数据
      lsm6dsv16x_acceleration_raw_get(&ctx, raw_accel);
      lsm6dsv16x_angular_rate_raw_get(&ctx, raw_gyro);
    
      // 转换为物理单位（g和dps）
      sensor_buffer[write_idx].accel[0] = raw_accel[0] / 32768.0 * 2.0;  // ±2g
      sensor_buffer[write_idx].accel[1] = raw_accel[1] / 32768.0 * 2.0;
      sensor_buffer[write_idx].accel[2] = raw_accel[2] / 32768.0 * 2.0;
    
      sensor_buffer[write_idx].gyro[0] = raw_gyro[0] / 32768.0 * 250.0;  // ±250dps
      sensor_buffer[write_idx].gyro[1] = raw_gyro[1] / 32768.0 * 250.0;
      sensor_buffer[write_idx].gyro[2] = raw_gyro[2] / 32768.0 * 250.0;
    
      sensor_buffer[write_idx].timestamp = HAL_GetTick();
    
      write_idx = (write_idx + 1) % BUFFER_SIZE;
  }

  // 获取缓冲区数据用于AI
  void getBufferData(float *output, int count) {
      int idx = (write_idx - count + BUFFER_SIZE) % BUFFER_SIZE;
      for (int i = 0; i < count; i++) {
          output[i*6 + 0] = sensor_buffer[idx].accel[0];
          output[i*6 + 1] = sensor_buffer[idx].accel[1];
          output[i*6 + 2] = sensor_buffer[idx].accel[2];
          output[i*6 + 3] = sensor_buffer[idx].gyro[0];
          output[i*6 + 4] = sensor_buffer[idx].gyro[1];
          output[i*6 + 5] = sensor_buffer[idx].gyro[2];
          idx = (idx + 1) % BUFFER_SIZE;
      }
  }

  typedef enum {
      MODE_SLEEP,      // 1 µA
      MODE_STANDBY,    // 10 µA
      MODE_NORMAL,     // 100 mA (160MHz)
      MODE_ACTIVE      // 150 mA (全功率)
  } PowerMode_t;

  void setPowerMode(PowerMode_t mode) {
      switch(mode) {
          case MODE_SLEEP:
              SystemClock_Config_MSI_16MHz();  // 16MHz
              HAL_PWREx_EnterSTOPMode();       // STOP模式
              break;

          case MODE_NORMAL:
              SystemClock_Config_PLL_160MHz(); // 160MHz
              break;
    
          case MODE_ACTIVE:
              // 已是最高频率
              break;
      }
  }

  // 根据用户活动自动切换
  void vPowerTask(void *pvParameters) {
      uint32_t inactivity_time = 0;

      while(1) {
          float accel_magnitude = sqrt(
              last_accel[0]*last_accel[0] +
              last_accel[1]*last_accel[1] +
              last_accel[2]*last_accel[2]
          );
    
          if (accel_magnitude < 0.2) {
              inactivity_time++;
              if (inactivity_time > 60) {  // 60秒无活动
                  setPowerMode(MODE_SLEEP);
              }
          } else {
              inactivity_time = 0;
              setPowerMode(MODE_NORMAL);
          }
    
          vTaskDelay(pdMS_TO_TICKS(1000));
      }
  }

  // 工作功耗: 150mA × 10min/小时 = 25mAh/小时

  // 假设白天活动6小时，晚上18小时待机

  3.1 气压传感器集成（方案B第一步）

  为什么要气压传感器？
  加速度传感器识别不出：爬楼梯 vs 走平地
  加气压传感器后：
    - 楼梯上升时：气压降低（高度增加）
    - 平地行走：气压恒定

  这样就能区分"平地步行"vs"楼梯步行"→更准确的活动强度评估

  集成步骤：
  // 添加气压传感器读取

      int32_t raw_pressure;
      float pressure_hPa;

      lps22df_pressure_raw_get(&ctx, &raw_pressure);
      pressure_hPa = raw_pressure / 4096.0;  // 转换为hPa
    
      // 计算高度变化（海平面参考）
      // h = 44330 * (1 - (P/P0)^(1/5.255))
      float height_change = calculateHeightChange(pressure_hPa);
    
      if (height_change > 0.5) {
          printf("上升楼梯\n");
          current_activity = CLIMBING_STAIRS;
      } else {
          printf("平地行走\n");
          current_activity = WALKING;
      }
  }

  // 修改AI评分逻辑
  void calculateScore(int action, activity_type_t activity) {
      if (action == WALKING) {
          if (activity == CLIMBING_STAIRS) {
              // 爬楼梯时给更高的活动强度系数
              difficulty_coef = 1.5;
          } else {
              difficulty_coef = 1.0;
          }
      }

      float final_score = base_score * difficulty_coef;
  }
